create or replace procedure swap_no
(
a in out number,
b in out number
)
as
begin

a := a+b;
b := a-b;
a := a-b;
DBMS_output.put_line('After Swapping:');
DBMS_output.put_line('A = ' || a);
DBMS_output.put_line('B = ' || b);
END;
/

----EXECUTION--------
variable a number;
exec:a:=5;

variable b number;
exec:b:=10;

exec swap_no(:a,:b);
---------------------




create or replace procedure largest_no
(
    a in number,
    b in number,
    c in number
)
as
begin
    if a > b and a > c then
        DBMS_output.put_line('A = ' || a);

    elsif b > a and b > c then
        DBMS_output.put_line('B = ' || b);

    else
        DBMS_output.put_line('C = ' || c);
	end if;
end;
/

-----EXECUTION-----
variable a number;
exec :a := 15;

variable b number;
exec :b := 25;

variable c number;
exec :c := 10;

exec largest_no(:a,:b,:c);
